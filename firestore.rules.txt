rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    function signedIn(){ return request.auth != null; }
    function isAdmin(){
      return signedIn()
        && exists(/databases/$(database)/documents/users/$(request.auth.uid))
        && get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == "admin";
    }

    match /users/{uid} {
      allow read: if signedIn() && request.auth.uid == uid;
      allow create: if signedIn() && request.auth.uid == uid;
      allow update: if isAdmin() || (signedIn() && request.auth.uid == uid);
    }

    match /activities_catalog/{id} {
      allow read: if true;
      allow write: if isAdmin();
    }

    match /user_logs/{id} {
      allow read, create: if signedIn() && request.resource.data.userId == request.auth.uid;
      allow update, delete: if isAdmin();
    }

    match /posts/{id} {
      allow read: if true;
      allow create: if signedIn();
      allow update, delete: if isAdmin();
    }

    match /post_comments/{id} {
      allow read: if true;
      allow create: if signedIn();
      allow update, delete: if isAdmin();
    }

    match /post_likes/{id} {
      allow read: if true;
      allow create: if signedIn();
      allow update, delete: if isAdmin();
    }

    match /chat_stories/{id} {
      allow read: if true;
      allow create: if signedIn();
      allow update, delete: if isAdmin();
    }

    match /tales/{id} {
      allow read: if true;
      allow write: if isAdmin();
    }

    match /connection_requests/{id} {
      allow read: if signedIn() && (resource.data.fromUid == request.auth.uid || resource.data.toUid == request.auth.uid);
      allow create: if signedIn();
      allow update: if signedIn();
      allow delete: if isAdmin();
    }

    match /connections/{id} {
      allow read: if signedIn();
      allow write: if isAdmin();
    }
  }
}
